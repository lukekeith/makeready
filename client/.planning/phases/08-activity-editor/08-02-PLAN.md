**1. Create `resources/js/components/admin/admin-activity-list/admin-activity-list.vue`:**

Create the directory first: `resources/js/components/admin/admin-activity-list/`

This component receives `programId`, `lessonId` as props and manages its own activity list. It imports domain store directly (same pattern as programs-section.vue).

```vue
<script setup lang="ts">
import { ref, watch } from 'vue'
import { GripVertical, Pencil, Trash2, Plus, RotateCcw } from 'lucide-vue-next'
import { VueDraggable } from 'vue-draggable-plus'
import { useProgramsDomain, ACTIVITY_TYPE_LABELS } from '../../../islands/admin-island/stores/domain/programs.domain'
import type { Activity } from '../../../islands/admin-island/stores/domain/programs.domain'
import { useProgramDetailUI } from '../../../islands/admin-island/stores/ui/program-detail.ui'
import { useActivityDetailUI } from '../../../islands/admin-island/stores/ui/activity-detail.ui'

const props = defineProps<{
  programId: string
  lessonId: string
}>()

const domain = useProgramsDomain()
const detailUI = useProgramDetailUI()
const activityUI = useActivityDetailUI()

// Local writable ref for vue-draggable-plus (same pattern as lesson reorder)
const localActivities = ref<Activity[]>([])

watch(
  () => detailUI.activitiesForExpandedLesson,
  (newActivities) => {
    localActivities.value = [...newActivities]
  },
  { immediate: true, deep: true }
)

// Add activity form state
const isAddFormOpen = ref(false)
const newActivityType = ref('READ')
const newActivityTitle = ref('')

const activityTypes = [
  { value: 'READ', label: 'Read' },
  { value: 'VIDEO', label: 'Video' },
  { value: 'USER_INPUT', label: 'Study' },
  { value: 'SOAP', label: 'SOAP' },
  { value: 'OIA', label: 'OIA' },
  { value: 'DBS', label: 'DBS' },
  { value: 'HEAR', label: 'HEAR' },
]

function openAddForm(): void {
  isAddFormOpen.value = true
  newActivityType.value = 'READ'
  newActivityTitle.value = ''
}

function closeAddForm(): void {
  isAddFormOpen.value = false
}

async function handleAdd(): Promise<void> {
  const title = newActivityTitle.value.trim()
  if (!title) return
  await domain.addActivity(props.programId, props.lessonId, newActivityType.value, title)
  closeAddForm()
}

async function handleDelete(): Promise<void> {
  const activity = detailUI.confirmDeleteActivity
  if (!activity) return
  try {
    await domain.deleteActivity(activity.id, props.programId, props.lessonId)
  } finally {
    detailUI.cancelDeleteActivity()
  }
}

async function handleReorder(): Promise<void> {
  const ids = localActivities.value.map((a) => a.id)
  await domain.reorderActivities(props.programId, props.lessonId, ids)
}

function handleActivityClick(activity: Activity): void {
  if (activityUI.editingActivityId === activity.id) {
    activityUI.closeEditor()
  } else {
    activityUI.openEditor(activity.id, props.programId, props.lessonId)
  }
}

function getTypeBadgeClass(activityType: string): string {
  return `AdminActivityList__type-badge AdminActivityList__type-badge--${activityType}`
}

function getStatusClass(status: string): string {
  return status === 'COMPLETE'
    ? 'AdminActivityList__status AdminActivityList__status--complete'
    : 'AdminActivityList__status AdminActivityList__status--pending'
}
</script>

<template>
  <div class="AdminActivityList">
    <!-- Activity list with drag-and-drop -->
    <template v-if="localActivities.length > 0">
      <VueDraggable
        v-model="localActivities"
        :animation="200"
        handle=".AdminActivityList__drag-handle"
        ghost-class="AdminActivityList__item--dragging"
        @end="handleReorder"
      >
        <div
          v-for="activity in localActivities"
          :key="activity.id"
          class="AdminActivityList__item"
          :class="{ 'AdminActivityList__item--editing': activityUI.editingActivityId === activity.id }"
          @click="handleActivityClick(activity)"
        >
          <span class="AdminActivityList__drag-handle" @click.stop>
            <GripVertical :size="14" />
          </span>
          <span :class="getTypeBadgeClass(activity.activityType)">
            {{ ACTIVITY_TYPE_LABELS[activity.activityType] || activity.activityType }}
          </span>
          <span class="AdminActivityList__title">
            {{ activity.title || '(untitled)' }}
          </span>
          <span :class="getStatusClass(activity.status)">
            {{ activity.status === 'COMPLETE' ? 'Complete' : 'Pending' }}
          </span>
          <div class="AdminActivityList__actions" @click.stop>
            <button
              class="AdminActivityList__action-btn"
              @click="handleActivityClick(activity)"
              title="Edit activity"
            >
              <Pencil :size="13" />
            </button>
            <button
              class="AdminActivityList__action-btn"
              @click="detailUI.requestDeleteActivity(activity.id)"
              title="Delete activity"
            >
              <Trash2 :size="13" />
            </button>
          </div>
        </div>
      </VueDraggable>
    </template>
    <div v-else class="AdminActivityList__empty">
      No activities yet. Add your first activity below.
    </div>

    <!-- Add activity form -->
    <div class="AdminActivityList__add-row">
      <template v-if="isAddFormOpen">
        <select v-model="newActivityType" class="AdminActivityList__type-select">
          <option v-for="t in activityTypes" :key="t.value" :value="t.value">{{ t.label }}</option>
        </select>
        <input
          v-model="newActivityTitle"
          class="AdminActivityList__add-title-input"
          placeholder="Activity title..."
          @keydown.enter="handleAdd"
          @keydown.escape="closeAddForm"
        />
        <button class="AdminActivityList__action-btn" @click="handleAdd" title="Add">
          <Plus :size="14" />
        </button>
        <button class="AdminActivityList__action-btn" @click="closeAddForm" title="Cancel">
          &times;
        </button>
      </template>
      <button v-else class="AdminActivityList__add-btn" @click="openAddForm">
        <Plus :size="14" /> Add Activity
      </button>
    </div>
  </div>
</template>
```
