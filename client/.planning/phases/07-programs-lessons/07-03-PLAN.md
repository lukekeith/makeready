Add handler functions for detail view:

```typescript
async function handleTogglePublish(): Promise<void> {
  if (!detailUI.currentProgram) return
  await domain.updateProgram(detailUI.currentProgram.id, {
    isPublished: !detailUI.currentProgram.isPublished,
  })
}

async function handleMetadataSave(payload: Record<string, any>): Promise<void> {
  detailUI.isSavingMetadata = true
  detailUI.metadataError = null
  try {
    await domain.updateProgram(route.params.id as string, payload)
  } catch (err: any) {
    detailUI.metadataError = err?.response?.data?.message ?? 'Failed to save'
  } finally {
    detailUI.isSavingMetadata = false
  }
}

async function handleCoverUpload(file: File): Promise<void> {
  detailUI.isUploadingCover = true
  try {
    await domain.uploadCoverImage(route.params.id as string, file)
  } finally {
    detailUI.isUploadingCover = false
  }
}

async function handleAddLesson(): Promise<void> {
  await domain.addLesson(route.params.id as string)
}

function startInlineEdit(lesson: Lesson): void {
  inlineEditId.value = lesson.id
  inlineEditTitle.value = lesson.title
}

async function saveInlineEdit(): Promise<void> {
  if (!inlineEditId.value) return
  await domain.updateLessonTitle(route.params.id as string, inlineEditId.value, inlineEditTitle.value)
  inlineEditId.value = null
  inlineEditTitle.value = ''
}

function cancelInlineEdit(): void {
  inlineEditId.value = null
  inlineEditTitle.value = ''
}

async function handleDeleteLesson(): Promise<void> {
  const lesson = detailUI.confirmDeleteLesson
  if (!lesson) return
  try {
    await domain.deleteLesson(route.params.id as string, lesson.id)
  } finally {
    detailUI.cancelDeleteLesson()
  }
}

async function handleReorder(): Promise<void> {
  const ids = localLessons.value.map((l) => l.id)
  await domain.reorderLessons(route.params.id as string, ids)
}
```
